#include <iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
void kt(unsigned a)
{
    int i,c=0,l=0;
    while(a>0)
    {
        i=a%10;
        a/=10;
        if(i%2==0)
        c++;
        else
        l++;
    }
    cout<<c<<" "<<l<<endl;
}
int main() {
    int a;
    unsigned b;
    cin>>a;
    for(int i=0;i<a;i++)
    {
    cin>>b;
    kt(b);
    }
    return 0;
}
