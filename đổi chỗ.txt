#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int a[1001][1001],n,m,x1,x2,y1,y2;
    cin>>n>>m;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        cin>>a[i][j];
    }
    cin>>x1>>y1>>x2>>y2;
    swap(a[x1-1][y1-1],a[x2-1][y2-1]);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        cout<<a[i][j]<<" ";
        cout<<endl;
    }
    return 0;
}
